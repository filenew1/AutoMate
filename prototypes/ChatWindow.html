def fibonacci(n):
    """
    计算斐波那契数列的第n项
    
    Args:
        n: 非负整数
        
    Returns:
        斐波那契数列的第n项
    """
    if n < 0:
        raise ValueError("n 必须是非负整数")
    elif n == 0:
        return 0
    elif n == 1:
        return 1
    
    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b
